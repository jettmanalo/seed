│       ├── typography/                 # Answer keys for typography bugs

Setup:
mkdir M-S2C-Framework && cd M-S2C-Framework

mkdir -p data/01_raw_seeds/{layering,layout,typography,spacing,visibility}
mkdir -p data/02_mutated_code/{layering,layout,typography,spacing,visibility}
mkdir -p data/03_screenshots/{layering,layout,typography,spacing,visibility}
mkdir -p data/04_ground_truth/{layering,layout,typography,spacing,visibility}

mkdir -p src/generators src/screenshotter

touch src/generators/{layering_mutator.py,layout_mutator.py,typography_mutator.py,spacing_mutator.py,visibility_mutator.py,tone_generator.py}
touch src/screenshotter/{renderer.py,capture.py}
touch requirements.txt
